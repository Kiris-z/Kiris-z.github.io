不定积分

$$
\int \tan xdx = -\ln |\cos x|+C\
\int \cot xdx = \ln |\sin x|+C\
\int \csc^2 x dx = -\cot x +C\
\int \sec x \tan x dx = \sec x +C\
\int \csc x \cot x dx = -\csc x +C\
\int \frac{dx}{a^2+x^2} = \frac{1}{a}\arctan \frac{x}{a}+C\
\int \frac{dx}{\sqrt{a^2-x^2}}=\arcsin \frac{x}{a}+C\
\int \frac{dx}{x^2-a^2} = \frac{1}{2a}\ln |\frac{x-a}{x+a}|+C\
\int \frac{dx}{\sqrt{x^2-a^2}}=\ln|x+\sqrt{x^2-a^2}|+C\
\int \frac{dx}{\sqrt{x^2+a^2}}=\ln(x+\sqrt{x^2+a^2})+C\
\int \sec x dx = \ln|\sec x+\tan x|+C\
\int \csc x dx = -\ln|\csc x+\cot x|+C\
\int \frac{1}{(x+a)(x+b)}dx = \frac{1}{b-a}\ln|\frac{x+a}{x+b}|+C(a\neq b)\
\int \ln(1+t)dt = (t+1)\ln(t+1)-t+C
$$

展开式

$$
\text{拉格朗日余项} R_n(x) = \frac{f^{n+1}(\xi)}{(n+1)!}(x-x_0)^{n+1}\
e^x=1+x+\frac{x^2}{2!}+\cdots +\frac{x^n}{n!}+o(x^n),(-\infty,+\infty)\
\sin x = x-\frac{x^3}{3!}+\cdots +(-1)^{n-1}\frac{x^{2n-1}}{(2n-1)!}+o(x^{2n-1}),(-\infty,+\infty)\
\cos x = 1-\frac{x^2}{2!}+\cdots +(-1)^{n}\frac{x^{2n}}{(2n)!}+o(x^{2n}),(-\infty,+\infty)\
\ln(1+x) = x-\frac{x^2}{2}+\cdots +(-1)^{n-1}\frac{x^{n}}{n}+o(x^{n}),(-1,1]\
-\ln(1-x) = x+\frac{x^2}{2}+\cdots +\frac{x^{n}}{n}+o(x^{n}),[-1,1)\
(1+x)^\alpha=1+\alpha x+\frac{\alpha(\alpha -1)}{2!}x^2+\cdots +\frac{\alpha(\alpha -1)\cdots(\alpha-n+1)}{n!}x^n+o(x^n),(-1,1),\text{区间端点展开式是否成立由}\alpha\text{决定}\

\frac{1}{1-x} = 1+x+x^2+\cdots +x^n+\cdots,(-1,1)\
\frac{1}{1+x} = 1-x+x^2-\cdots +(-1)^nx^n+\cdots,(-1,1)\
$$

三角函数

$$
\cos(\arcsin x)=\sqrt{1-x^2}\
\arcsin(-x) = -\arcsin x\
\arccos(-x)=\pi-\arccos x\
\arctan(-x)=-\arctan x\
arccot(-x) = \pi-arccot x\
\arcsin x+\arccos x =\arctan x+arccot x= \frac{\pi}{2}\
\sin(\arcsin x)=\cos(\arccos x)=\tan(\arctan x)=\cot(arccot x) = x\
\text{当} x\in[-\frac{\pi}{2},\frac{\pi}{2}]\text{有}\arcsin(\sin x) = x\
\text{当} x\in[0,\pi]\text{有}\arccos(\cos x) = x\
\text{当} x\in(-\frac{\pi}{2},\frac{\pi}{2})\text{有}\arctan(\tan x) = x\
\text{当} x\in(0,\pi)\text{有}arccot(cot\ x) = x\
x>0,\arctan x+\arctan \frac{1}{x}=\frac{\pi}{2}\
\text{若}(\arctan x+\arctan y)\in(-\frac{\pi}{2},\frac{\pi}{2}),\text{则}\arctan x+\arctan y=\arctan(\frac{x+y}{1-xy})\
1+\sin(x) = (\sin(\frac{x}{2})+\cos(\frac{x}{2}))^2\
1-\sin(x) = (\sin(\frac{x}{2})-\cos(\frac{x}{2}))^2\
$$
